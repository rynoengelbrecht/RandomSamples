SELECT TOP 100
    w.name,
    w.workflowid,
    w.category,
    w.primaryentity,
    w.statecode,
    w.type,
    w.createdon
FROM
    workflow AS w
WHERE
    w.category IN (5, 6)   -- 5 = cloud flow, 6 = desktop flow
    AND w.type = 1         -- definitions, not instances
ORDER BY
    w.createdon DESC;
