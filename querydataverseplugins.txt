SELECT
    w.name           AS FlowName,
    w.workflowid,
    w.category,
    w.primaryentity,
    w.statecode,
    w.clientdata
FROM
    workflow AS w
WHERE
    w.category = 5           -- Cloud flows
    AND w.type = 1           -- Definitions
    AND LOWER(w.clientdata) LIKE '%account%';
